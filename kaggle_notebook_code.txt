class StudyAgent:
    def __init__(self, model):
        self.model = model

    def explain_topic(self, topic):
        prompt = f"Explain '{topic}' in simple language."
        return self.model.generate_content(prompt).text

    def generate_notes(self, topic):
        prompt = f"Give me short bullet-point study notes for: {topic}"
        return self.model.generate_content(prompt).text

    def generate_mcq(self, topic):
        prompt = f"""
        Generate 5 MCQs for the topic '{topic}'.
        Format:
        Q1) question
        A) option
        B) option
        C) option
        D) option
        Correct Answer: A
        """
        return self.model.generate_content(prompt).text

agent = StudyAgent(model)
